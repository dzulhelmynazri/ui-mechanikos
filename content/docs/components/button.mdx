## Usage

```tsx
import { Button, ButtonArrow, buttonVariants } from "@/components/ui/button"
```

```tsx
<Button
  variant="primary | mono | destructive | secondary | outline | dashed | ghost | dim | foreground | inverse"
  appearance="default | ghost"
  size="sm | md | lg | icon"
  shape="default | circle"
  mode="default | icon | link | input"
  autoHeight
  placeholder
  underline="solid | dashed"
  underlined="solid | dashed"
>
  Custom icon
  <ButtonArrow icon={ChevronDown} />
</Button>
```

## Link

You can use the `asChild` prop to make another component look like a button. Here's an example of a link that looks like a button.

```tsx showLineNumbers
import { Link } from "next/link"

import { Button } from "@/components/ui/button"

export function LinkAsButton() {
  return (
    <Button asChild>
      <Link href="/login">Login</Link>
    </Button>
  )
}
```
